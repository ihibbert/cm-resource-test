                    const button = document.createElement('button');
                    button.textContent = category;
                    button.className = `w-full text-left px-4 py-2 rounded-md transition-colors duration-200 hover:bg-[#e9e2dc] ${category === activeCategory ? 'active-nav' : 'text-slate-700'}`;
                    button.onclick = () => {
                        activeCategory = category;
                        renderContent();
                        updateNavs();
                        if(window.innerWidth < 768) {
                            mobileSidebarContainer.classList.add('hidden');
                        }
                    };
                    li.appendChild(button);
                    listElement.appendChild(li);
                });
            }
            
            function renderContent(filter = '') {
                const categoryData = guideData[activeCategory];
                currentCategoryTitle.textContent = activeCategory;
                categoryIntroduction.textContent = categoryData.introduction;
                linkGrid.innerHTML = '';
                
                const filteredLinks = categoryData.links.filter(link => 
                    link.title.toLowerCase().includes(filter.toLowerCase()) || 
                    link.summary.toLowerCase().includes(filter.toLowerCase())
                );

                if (filteredLinks.length === 0) {
                    linkGrid.innerHTML = `<p class="text-slate-500 col-span-full text-center">No results found.</p>`;
                } else {
                    filteredLinks.forEach(link => {
                        const card = document.createElement('div');
                        card.className = 'bg-white border border-gray-200 rounded-xl p-6 flex flex-col hover:shadow-lg hover:border-blue-300 transition-all duration-300';
                        card.innerHTML = `
                            <h3 class="text-lg font-semibold text-slate-800 mb-2">${link.title}</h3>
                            <p class="text-slate-600 flex-grow mb-4">${link.summary}</p>
                            <a href="${link.url}" target="_blank" rel="noopener noreferrer" class="mt-auto inline-block bg-blue-600 text-white text-center font-medium py-2 px-4 rounded-lg hover:bg-blue-700 transition-colors duration-200">
                                Visit Site &rarr;
                            </a>
                        `;
                        linkGrid.appendChild(card);
                    });
                }
            }

            function updateNavs() {
                document.querySelectorAll('#category-list button, #mobile-category-list button').forEach(button => {
                    if (button.textContent === activeCategory) {
                        button.classList.add('active-nav');
                         button.classList.remove('text-slate-700');
                    } else {
                        button.classList.remove('active-nav');
                        button.classList.add('text-slate-700');
                    }
                });
            }
            
            searchInput.addEventListener('input', (e) => {
                renderContent(e.target.value);
            });

            menuButton.addEventListener('click', () => {
                mobileSidebarContainer.classList.remove('hidden');
            });

            closeMenuButton.addEventListener('click', () => {
                mobileSidebarContainer.classList.add('hidden');
            });
            
            mobileSidebarContainer.addEventListener('click', (e) => {
                if(e.target === mobileSidebarContainer){
                    mobileSidebarContainer.classList.add('hidden');
                }
            });


            // Initial Render
            renderCategories(categoryList);
            renderCategories(mobileCategoryList);
            renderContent();
        });
    </script>
